        <script>
            // 变量定义（let/const/var）
            let name = "李青";       // 字符串 string
            const age = 20;          // 数字 number（常量，不可改）
            var isStudent = true;    // 布尔 boolean
            let hobby = ["编程", "阅读"]; // 数组 Array
            let info = {name: name, age: age}; // 对象 object

            // 打印变量类型
            console.log("name 类型:", typeof name);
            console.log("age 类型:", typeof age);
            console.log("hobby 类型:", typeof hobby);
        </script>
        <p>在浏览器控制台可查看变量类型输出</p >

    <!-- 2. 表达式 -->
    <div class="section">
        <h2>2. 表达式运算</h2>
        <script>
            let a = 10, b = 3;
            let sum = a + b;    // 算术表达式
            let isGt = a > b;   // 比较表达式
            let logic = isGt && true; // 逻辑表达式

            console.log("10+3 =", sum);
            console.log("10>3 =", isGt);
            console.log("true&&true =", logic);
        </script>
        <p>在浏览器控制台可查看表达式运算结果</p >
    </div>

    <!-- 3. 流程控制 -->
    <div class="section">
        <h2>3. 流程控制语句</h2>
        <script>
            // if-else 条件判断
            let score = 85;
            if (score >= 90) {
                console.log("优秀");
            } else if (score >= 80) {
                console.log("良好");
            } else {
                console.log("合格");
            }

            // for 循环
            for (let i = 1; i <= 5; i++) {
                console.log("for循环:", i);
            }
        </script>
        <p>在浏览器控制台可查看流程控制执行结果</p >
    </div>

    <!-- 4. 函数 -->
    <div class="section">
        <h2>4. 函数定义与调用</h2>
        <script>
            // 普通函数
            function add(num1, num2) {
                return num1 + num2;
            }
            // 箭头函数（ES6+）
            const multiply = (a, b) => a * b;

            console.log("add(5,8) =", add(5,8));
            console.log("multiply(4,6) =", multiply(4,6));
        </script>
        <p>在浏览器控制台可查看函数调用结果</p >
